if Statement

it will run the block code {}, if the condition has met


            #define _CRT_SECURE_NO_WARNINGS
            #include 

            int main(){
                int x = 0;

                if(x == 0){
                    printf("if x is equal to 0, meet the condition, it will run this printf.\n");
                }else if(x == 1){
                    printf("else if x is equal to 1, meet the condition, it will run this printf.\n");
                }else{
                    printf("if x is not equal to 0 and it is not equal to 1, it will goes into else statement\n");
                }

                return 0;
            }

while loop

it will keep running the code to infinite times, until the condition has met


            #define _CRT_SECURE_NO_WARNINGS
            #include 

            int main(){
                int x = 0;

                while(x < 3){
                    printf("Run infinite until x is equal or greater than 3: %d.\n", x);
                    x++; //increment x;
                }

                return 0;
            }

do while loop

Same concept as while loop, but do while loop will run at least once.


            #define _CRT_SECURE_NO_WARNINGS
            #include 

            int main(){
                int x = 0;

                while(x < 3){
                    printf("Run infinite until x is equal or greater than 3: %d.\n", x);
                    x++; //increment x;
                }

                return 0;
            }

for loop

Using for loop when you know how many times it should run


            #define _CRT_SECURE_NO_WARNINGS
            #include 

            int main(){
                int x = 1;

                for(int i = 0; i < 3; i++){
                    printf("%d. Printing x.\n", i); //it will print 3 times
                }

                return 0;
            }
